36void
37exif_mnote_data_construct (ExifMnoteData *d, ExifMem *mem)
38{
39 if (!d || !mem) return;
40 if (d->priv) return;
41 d->priv = exif_mem_alloc (mem, sizeof (ExifMnoteDataPriv));
42 if (!d->priv) return;

46 d->mem = mem;

56static void
57exif_mnote_data_free (ExifMnoteData *d)
58{
59 ExifMem *mem = d ? d->mem : NULL;
60
61 if (!d) return;
62 if (d->priv) {
63 if (d->methods.free) d->methods.free (d);
64 exif_mem_free (mem, d->priv);
65 d->priv = NULL;
66 }
67 exif_log_unref (d->log);
68 exif_mem_free (mem, d);
69 exif_mem_unref (mem);
70}

72void

77 if (!d->priv->ref_count)
