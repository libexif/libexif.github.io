213  if (table[i].tag == t) return (_(table[i].title));

217 const char *
218 mnote_olympus_tag_get_description (MnoteOlympusTag t)
219 {
220  unsigned int i;
221 
222  for (i = 0; i < sizeof (table) / sizeof (table[0]); i++)
223  if (table[i].tag == t) {
224  if (!table[i].description || !*table[i].description)
225  return "";
226  (void) bindtextdomain (GETTEXT_PACKAGE, LOCALEDIR);
227  return _(table[i].description);
228  }
229  return NULL;
